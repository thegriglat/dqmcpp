92 std::vector<dqmcpp::ECAL::ChannelData> parse(Document& j) {
93  using namespace std;
94  // check validity of json
95  const auto data2d = parse2D(j);
96  if (data2d.size() == 0)
97  return std::vector<ECAL::ChannelData>();
98  const auto iz = getECALDetector(j["hist"]["title"].GetString());
99  vector<ECAL::ChannelData> channel_data;
100  const auto& xaxis = j["hist"]["xaxis"];
101  const std::string xtitle = xaxis["title"].GetString();
102  const char xsign = (xtitle.find('-') != xtitle.npos) ? -1 : 1;
103  for (auto& datapoint : data2d) {
104  const int x = std::trunc(datapoint.base.x);
105  const int y = std::trunc(datapoint.base.y);
106  int ix_iphi;
107  int iy_ieta;
108  if (iz == 0) {
109  // EB
110  // in case of barrel we have to swap x and y values
111  ix_iphi = std::abs(y) + 1;
112  iy_ieta = (x + 1) * xsign;
113  } else {
114  // EE
115  ix_iphi = x + 1;
116  iy_ieta = y + 1;
117  }
118  ECAL::Channel c(ix_iphi, iy_ieta, iz);
119  channel_data.push_back(ECAL::ChannelData(c, datapoint.value));
120  }
121  return channel_data;
122 }
