85  all_is_zero = false;

88  }

90  tt_to_remove.push_back({tt.base.tt, tt.base.iz, tt.base.tcc});
91  }
92  }
93  for (auto& ttremove : tt_to_remove) {
94  for (auto& run : rundata) {
95  auto it = std::find_if(

97  return ttremove[0] == e.base.tt && ttremove[1] == e.base.iz &&
98  ttremove[2] == e.base.tcc;
99  });
100  if (it == run.data.end())
101  continue;
102  run.data.erase(it);
103  }
104  }
105  return rundata;
