35 std::string urlencode(const std::string& value) {
36  ostringstream escaped;
37  escaped.fill('0');
38  escaped << hex;
39  for (auto c : value) {
40  // Keep alphanumeric and other accepted characters intact
41  if (isalnum(c) || c == '-' || c == '_' || c == '.' || c == '~' ||
42  c == '/' || c == '+') {
43  escaped << c;
44  continue;
45  }
46  // Any other characters are percent-encoded
47  escaped << uppercase;
48  escaped << '%' << setw(2) << int((unsigned char)c);
49  escaped << nouppercase;
50  }
51  return escaped.str();
52 }

61 int maxDifference(const string& s1, const string& s2) {
62  const int s1len = s1.length();
63  const int s2len = s2.length();

81  const int maxDiff,
82  std::function<std::string(const T&)> getter) {
83  vector<vector<T>> groups;
84  vector<T> skiplist;
85  for (auto it = list.begin(); it != list.end(); ++it) {
86  const auto field_it = getter(*it);
87  const auto it_vlist = dqmcpp::common::split(field_it, "/");
88  const auto it_dataset_part = it_vlist.at(1);
89 
90  vector<T> group;
91  for (auto jit = it; jit != list.end(); ++jit) {
92  if (dqmcpp::common::has(skiplist, *jit))
93  continue;
94  const auto field_jit = getter(*jit);
95  const auto jit_vlist = dqmcpp::common::split(field_jit, "/");
96  const auto jit_dataset_part = jit_vlist.at(1);
97  const auto diff = maxDifference(field_it, field_jit);
98 
99  if (diff <= maxDiff && it_dataset_part == jit_dataset_part) {
100  group.push_back(*jit);
101  skiplist.push_back(*jit);
102  }
103  }
104  groups.push_back(group);
105  }
106  // remove empty groups
107  groups.erase(
108  std::remove_if(groups.begin(), groups.end(),
109  [](const vector<T>& group) { return group.size() == 0; }),
110  groups.end());
111  // sort
112  for (auto& gr : groups)
113  std::sort(gr.begin(), gr.end(), [getter](const T& a, const T& b) {
114  return getter(a) < getter(b);
115  });
116  return groups;
117 };
